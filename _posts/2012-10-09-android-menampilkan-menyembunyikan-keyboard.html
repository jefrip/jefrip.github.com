sebelumnya import terlebih dahulu

{% highlight java startinline linenos %} InputMethodManager imm = (InputMethodManager)getSystemService(Context.INPUT_METHOD_SERVICE); imm.hideSoftInputFromWindow(editText.getWindowToken(), 0); {% endhighlight %}
